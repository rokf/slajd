function load_images()
  for _,s in ipairs(data) do
    for _,l in ipairs(s) do
      if l.type == "image" then
        if images[l[1]] == nil then
          images[l[1]] = cairo.ImageSurface.create_from_png(l[1])
        end
      end
    end
  end
end
